一、DES

DES 算法为密码体制中的对称密码体制，又被称为美国数据加密标准。

DES 是一个分组加密算法，分组长度为 64 比特，其中有 8 位（第8、16、24、32、40、48、56、64位）是奇偶校验位，因此实际长度为 56 比特。因为它使用的 56 位密钥过短，现已被 AES 替代。

'''
DES Functions
'''
def padding(num, total):
    return format(num, '0%db' % total)


def join(list_of_str):
    return ''.join(list_of_str)


def left_shift(str_bin, total=1):
    return str_bin[total:] + str_bin[0:total]


def str_bin_to_hex(str_bin):
    partial = int(len(str_bin)/4)
    return join([format(int(str_bin[4*i:4*(i+1)] ,2), 'x') for i in range(partial)])


def str_bin_to_str_ascii(str_bin):
    partial = int(len(str_bin)/8)
    return join([format(int(str_bin[8*i:8*(i+1)] ,2), 'c') for i in range(partial)])


def hex_to_str_bin(hexa):
    return join([padding(int(c, 16), 4) for c in hexa])


def str_to_str_bin(txt):
    return join([padding(ord(c), 8) for c in txt])

#重排列
def permutate(str_bin, table):
    return join([str_bin[i-1] for i in table])

def xor(sb1, sb2):
    pad = len(sb1) if len(sb1) > len(sb2) else len(sb2)
    sb_1 = int(sb1, 2)
    sb_2 = int(sb2, 2)
    result = sb_1 ^ sb_2
    return padding(result, pad)

#明文多于64位时，以0填充
def split_binary(str_bin):
    list_text = []
    divisor, remainder = divmod(len(str_bin), 64)

    for i in range(divisor):
        list_text.append(str_bin[i*64:(i+1)*64])

    if remainder > 0:
        list_text.append(str_bin[-remainder:] + '0' * (64-remainder))

    return list_text


class DES(object):


    def __init__(self, hex_mode=False):
        self.hex_mode = hex_mode

    #密钥置换
    def generate_subkeys(self, key):
        if self.hex_mode:
            bin_key = hex_to_str_bin(key)
        else:
            bin_key = str_to_str_bin(key)

        Cn = []
        Dn = []
        Kn = []
        KP = permutate(bin_key, PC_1)

        Cn.append(KP[:28])
        Dn.append(KP[28:])

        #16轮密钥处理
        for i in range(16):
            Cn.append(left_shift(Cn[i], MOVE_TIMES[i]))
            Dn.append(left_shift(Dn[i], MOVE_TIMES[i]))
            CnDn = Cn[i+1] + Dn[i+1]
            Kn.append(join([CnDn[i-1] for i in PC_2]))

        return Kn

    def encrypt(self, message, key):
        if self.hex_mode:
            bin_key = hex_to_str_bin(key)
            #bin_msg = hex_to_str_bin(message)
        else:
            bin_key = str_to_str_bin(key)
            bin_msg = str_to_str_bin(message)

        partials = split_binary(bin_msg)
        partials_encoded = []

        subkeys = self.generate_subkeys(key)

        for m in partials:
            partials_encoded.append(self.encode(m, subkeys))

        encoded = join(partials_encoded)

        if self.hex_mode:
            encoded = str_bin_to_hex(encoded)
        else:
            encoded = str_bin_to_str_ascii(encoded)

        return encoded


    def decrypt(self, cipher, key):
        if self.hex_mode:
            bin_key = hex_to_str_bin(key)
            bin_msg = hex_to_str_bin(cipher)
        else:
            bin_key = str_to_str_bin(key)
            bin_msg = str_to_str_bin(cipher)

        partials = split_binary(bin_msg)
        partials_encoded = []

        subkeys = self.generate_subkeys(key)

        for m in partials:
            partials_encoded.append(self.encode(m, subkeys[::-1]))

        encoded = join(partials_encoded)

        if self.hex_mode:
            encoded = str_bin_to_hex(encoded)
        else:
            encoded = str_bin_to_str_ascii(encoded)

        return encoded.replace(chr(0), '')


    def encode(self, str_bin, Kn):
        bip = permutate(str_bin, IP_Table)

        Ln = []
        Rn = []

        Ln.append(bip[:32])
        Rn.append(bip[32:])

        for i in range(1,17):
            Ln.append(Rn[i-1])
            Rn.append(xor(Ln[i-1], self.fn(Rn[i-1], Kn[i-1])))

        RL = Rn[16] + Ln[16]

        return permutate(RL, IP_1_Table)


    def fn(self, ri, ki):
        ei = permutate(ri, E_Table)
        ei = xor(ki, ei)

        result = ''
        partials = [ei[6*i:6*(i+1)] for i in range(8)]

        for i, part in enumerate(partials):
            row = int(part[0] + part[-1], 2)
            col = int(part[1:5], 2)
            num = S[i][row][col]
            result += padding(num, 4)

        result = permutate(result, P_Table)

        return result
